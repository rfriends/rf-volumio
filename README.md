rfriends_volumioはvolumio環境でrfriends3を動作させるスクリプトです  
  
cd ~/  
sudo apt install git  
rm -rf rfriends_volumio
git clone https://github.com/rfriends/rfriends_volumio.git  
cd rfriends_volumio  
sh rfriends3_volumio.sh
